(module 
    (import "IO" "getpin" (func $getpin (param $pin i8) (result i8)))
    (import "IO" "setpin" (func $setpin (param $pin i8)))
(func (export "run")
(local $V1 i32)
(i32.const 12) ;;input port number

(local.set $V1)

(local $V2 i32)
(i32.const 23)
(local.set $V2)

(local.get $V1)

(call $setpin)
)
